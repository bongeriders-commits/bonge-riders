rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {

    function isAdmin() {
      return request.auth != null
        && request.auth.token.firebase.sign_in_provider == 'password';
    }
    function isAnyAuth() {
      return request.auth != null;
    }
    // A caller is a super admin if their email is listed in tracker/superAdmins,
    // OR if no admins have been configured yet at all (first-time setup bootstrap,
    // matching the client-side isSuperAdminUser() fallback in index.html).
    function isSuperAdmin() {
      let adminsDoc = /databases/$(database)/documents/tracker/admins;
      let noAdminsYet = !exists(adminsDoc)
        || !('list' in get(adminsDoc).data)
        || get(adminsDoc).data.list.size() == 0;
      let superAdminsDoc = /databases/$(database)/documents/tracker/superAdmins;
      return isAdmin() && (
        noAdminsYet ||
        (exists(superAdminsDoc) && request.auth.token.email in get(superAdminsDoc).data.emails)
      );
    }

    match /tracker/superAdmins {
      allow read: if isAnyAuth();
      allow write: if isAdmin();
    }

    // Administrator activity log — append-only, undeletable audit trail.
    match /activityLog/{entryId} {
      allow create: if isAdmin();
      allow read: if isSuperAdmin();
      allow update, delete: if false;
    }

    match /tracker/members {
      allow read, write: if isAdmin();
    }
    match /tracker/payments {
      // TODO(phase 2): this single doc holds every member's payment history.
      // Anonymous sessions can currently read all of it. Kept as isAnyAuth()
      // for now because member.html needs it for the portal, but this should
      // move to per-member subcollections (tracker/payments/{phone}) once the
      // phone-claim login (see memberPins below) ships, so this can become
      // owner-scoped instead of all-or-nothing.
      allow read: if isAnyAuth();
      allow write: if isAdmin();
    }
    match /tracker/income {
      allow read, write: if isAdmin();
    }
    match /tracker/spending {
      allow read: if isAnyAuth();
      allow write: if isAdmin();
    }
    match /tracker/dashboard {
      allow read: if isAnyAuth();
      allow write: if isAdmin();
    }
    match /tracker/settings {
      allow read: if isAnyAuth();
      allow write: if isAdmin();
    }
    match /tracker/admins {
      allow read, write: if isAdmin();
    }

    // ── PINs ──
    // SECURITY: previously readable by any anonymous session, which let
    // anyone dump every member's PIN in one query. Client-side PIN checks
    // must be replaced by the server-side /api/verify-pin endpoint (mints a
    // custom token). Once that ships, no client ever needs to read this
    // document directly, so lock it to admins only.
    match /tracker/memberPins {
      allow read, write: if isAdmin();
    }

    match /tracker/memberDetails {
      // TODO(phase 2): same shape problem as payments — one doc, all
      // members. Move to tracker/memberDetails/{phone} subcollection and
      // gate read on request.auth.token.phone == phone once phone-claim
      // login ships. Until then this stays isAnyAuth() so the member
      // portal keeps working.
      allow read:  if isAnyAuth();
      allow write: if isAdmin();
    }
    match /tracker/memberPhotos {
      allow read, write: if isAnyAuth();
    }
    match /tracker/saccoSettings {
      allow read:  if isAnyAuth();
      allow write: if isAdmin();
    }
    match /tracker/publicInfo {
      allow read:  if isAnyAuth();
      allow write: if isAdmin();
    }
    match /tracker/contactInfo {
      allow read:  if isAnyAuth();
      allow write: if isAdmin();
    }
    match /tracker/payment_confirmations {
      allow read:   if isAdmin();
      allow create: if isAnyAuth();
      allow update, delete: if isAdmin();
    }

    // ── Welfare applications ──
    // SECURITY FIX: previously ANY authenticated (incl. anonymous) user
    // could read or write ANY welfare_<phone> doc, and doc IDs are just
    // the member's phone digits — trivially guessable. Now scoped to the
    // owning member (via the `phone` custom claim minted by
    // /api/verify-pin.js on login) or an admin. Members who haven't logged
    // in through the new PIN endpoint yet (still on a bare anonymous
    // session with no phone claim) fall through to "no access" rather than
    // "everyone's welfare data," which is the safe direction to fail in.
    match /tracker/{docId} {
      allow read, write: if docId.matches('welfare_[0-9]+') && (
        isAdmin() ||
        (isAnyAuth() && 'phone' in request.auth.token
          && docId == 'welfare_' + request.auth.token.phone)
      );
    }

    // ── Projects ──
    match /projectItems/{itemId} {
      allow read: if isAnyAuth();
      allow create, update, delete: if isAdmin();
    }
    match /projectVotes/{voteId} {
      // voteId is "<projectId>_<uid>" — a member can only write their own
      // vote doc, never anyone else's.
      allow read: if isAnyAuth();
      allow create, update: if isAnyAuth()
        && voteId == request.resource.data.projectId + '_' + request.auth.uid;
      allow delete: if isAnyAuth()
        && voteId.split('_')[voteId.split('_').size() - 1] == request.auth.uid;
    }
    match /projectComments/{commentId} {
      allow read: if isAnyAuth();
      allow create: if isAnyAuth()
        && request.resource.data.uid == request.auth.uid;
      // Owner can delete their own comment; admins can delete any.
      allow delete: if isAdmin()
        || (isAnyAuth() && resource.data.uid == request.auth.uid);
      allow update: if false;
    }

    // ── Gallery ──
    match /galleryItems/{itemId} {
      allow read: if isAnyAuth();
      allow create, update, delete: if isAdmin();
    }

    // ── Contact / Messages ──
    match /contactMessages/{msgId} {
      // Anyone can submit a message; only admins can read the inbox or
      // manage/delete messages (previously messages.html deleted with no
      // rule backing it at all — it was silently falling through to the
      // deny-all catch-all).
      allow create: if isAnyAuth();
      allow read, update, delete: if isAdmin();
    }

    // Push notification device tokens.
    match /pushTokens/{token} {
      allow read: if false;
      allow write: if isAnyAuth() && token == request.resource.data.token;
      allow delete: if isAnyAuth();
    }

    match /{document=**} {
      allow read, write: if false;
    }
  }
}
